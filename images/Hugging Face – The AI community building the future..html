import pandas as pd
from sklearn.ensemble import RandomForestClassifier
import joblib

def analyze_responses(responses):
# Map responses from text to numerical values
mapping = {"Very Bad": 1, "Bad": 2, "Satisfactory": 3, "Good": 4, "Very Good": 5}
numeric_responses = {k: mapping[v] for k, v in responses.items()}
df = pd.DataFrame([numeric_responses])

# Load the trained model and predict
model = joblib.load('model.pkl')
prediction = model.predict(df)
return {'result': prediction[0]}

def train_model():
# Training data matching new options
data = {
'q1': [1, 2, 3, 4, 5],
'q2': [2, 3, 4, 5, 1],
'q3': [3, 4, 5, 1, 2],
'q4': [4, 5, 1, 2, 3],
'q5': [5, 1, 2, 3, 4],
'q6': [1, 2, 3, 4, 5],
'q7': [2, 3, 4, 5, 1],
'q8': [3, 4, 5, 1, 2],
'q9': [4, 5, 1, 2, 3],
'q10': [5, 1, 2, 3, 4],
'result': ['Analytical Thinker', 'Creative Leader', 'Organized Planner', 'Team Player', 'Tech Enthusiast']
}
df = pd.DataFrame(data)
X = df.drop(columns=['result'])
y = df['result']

# Train the RandomForest model
model = RandomForestClassifier(n_estimators=100, random_state=42)
model.fit(X, y)
joblib.dump(model, 'model.pkl')
print("Model trained and saved as model.pkl ‚úÖ")

if __name__ == '__main__':
train_model()
